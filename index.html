import pygame import sys import time # ====================== # JOGO: MOTA 2D - PASSA OBSTÁCULOS # Versão básica para GitHub # 8 níveis com dificuldade crescente # Usa apenas pygame (sem imagens externas) # ====================== pygame.init() # Configurações da janela LARGURA = 800 ALTURA = 600 tela = pygame.display.set_mode((LARGURA, ALTURA)) pygame.display.set_caption("Mota 2D - Passa os Obstáculos!") relogio = pygame.time.Clock() fonte = pygame.font.SysFont(None, 48) fonte_pequena = pygame.font.SysFont(None, 32) # Cores CEU = (135, 206, 235) CHAO_COR = (34, 139, 34) PRETO = (0, 0, 0) BRANCO = (255, 255, 255) VERMELHO = (200, 0, 0) CINZA = (100, 100, 100) # Variáveis do jogo CHAO_Y = 480 VELOCIDADE_BASE = 6 class Mota: def __init__(self): self.x = 150 # posição fixa (runner style) self.y = CHAO_Y - 60 self.largura = 70 self.altura = 40 self.vel_y = 0 self.no_chao = True def pular(self): if self.no_chao: self.vel_y = -18 self.no_chao = False def atualizar(self): self.vel_y += 1.2 # gravidade self.y += self.vel_y # Colisão com o chão if self.y >= CHAO_Y - self.altura: self.y = CHAO_Y - self.altura self.vel_y = 0 self.no_chao = True def desenhar(self, tela): # Corpo da mota (retângulo cinza) pygame.draw.rect(tela, CINZA, (self.x, self.y + 15, self.largura, 25)) # Roda traseira pygame.draw.circle(tela, PRETO, (self.x + 15, self.y + 38), 12) # Roda dianteira pygame.draw.circle(tela, PRETO, (self.x + 55, self.y + 38), 12) # Banco / piloto (triângulo vermelho) pontos = [(self.x + 20, self.y + 15), (self.x + 45, self.y + 10), (self.x + 60, self.y + 25)] pygame.draw.polygon(tela, VERMELHO, pontos) # Guidão pygame.draw.line(tela, PRETO, (self.x + 55, self.y + 18), (self.x + 70, self.y + 8), 6) class Obstaculo: def __init__(self, x): self.x = x self.y = CHAO_Y - 35 self.largura = 45 self.altura = 35 def atualizar(self, velocidade): self.x -= velocidade def desenhar(self, tela): # Caixa de obstáculo (vermelho com "espinhos") pygame.draw.rect(tela, VERMELHO, (self.x, self.y, self.largura, self.altura)) # Espinhos simples pygame.draw.line(tela, PRETO, (self.x + 10, self.y), (self.x + 10, self.y - 12), 4) pygame.draw.line(tela, PRETO, (self.x + 22, self.y), (self.x + 22, self.y - 12), 4) pygame.draw.line(tela, PRETO, (self.x + 34, self.y), (self.x + 34, self.y - 12), 4) def iniciar_nivel(nivel): """Gera os obstáculos para o nível atual""" global obstaculos, distancia, velocidade obstaculos = [] distancia = 0 # Dificuldade crescente velocidade = VELOCIDADE_BASE + nivel * 0.8 num_obstaculos = 4 + nivel * 2 # nível 1 = 6, nível 8 = 20 espacamento = max(220 - nivel * 12, 120) # ficam mais próximos x_inicial = LARGURA + 100 for i in range(num_obstaculos): x = x_inicial + i * espacamento obstaculos.append(Obstaculo(x)) # Distância necessária para completar o nível return 1800 + nivel * 450 # ====================== # VARIÁVEIS DO JOGO # ====================== mota = Mota() obstaculos = [] distancia = 0 nivel_atual = 0 distancia_alvo = iniciar_nivel(0) estado = "jogando" # jogando, game_over, vitoria, nivel_completo tempo_mensagem = 0 # Loop principal while True: for evento in pygame.event.get(): if evento.type == pygame.QUIT: pygame.quit() sys.exit() if evento.type == pygame.KEYDOWN: if estado == "jogando" and evento.key == pygame.K_SPACE: mota.pular() # Reiniciar nível com R if estado == "game_over" and evento.key == pygame.K_r: distancia_alvo = iniciar_nivel(nivel_atual) estado = "jogando" mota = Mota() # Sair com ESC if evento.key == pygame.K_ESCAPE: pygame.quit() sys.exit() # ====================== # LÓGICA DO JOGO # ====================== if estado == "jogando": mota.atualizar() # Atualiza obstáculos for obs in obstaculos[:]: obs.atualizar(velocidade) if obs.x < -100: obstaculos.remove(obs) # Colisão mota <-> obstáculo if (mota.x + mota.largura > obs.x and mota.x < obs.x + obs.largura and mota.y + mota.altura > obs.y and mota.y < obs.y + obs.altura): estado = "game_over" tempo_mensagem = time.time() distancia += velocidade # Verifica se passou o nível if distancia >= distancia_alvo and len(obstaculos) == 0: nivel_atual += 1 if nivel_atual >= 8: estado = "vitoria" tempo_mensagem = time.time() else: estado = "nivel_completo" tempo_mensagem = time.time() distancia_alvo = iniciar_nivel(nivel_atual) # ====================== # DESENHO # ====================== tela.fill(CEU) # céu # Chão pygame.draw.rect(tela, CHAO_COR, (0, CHAO_Y, LARGURA, ALTURA - CHAO_Y)) # Desenha mota e obstáculos if estado != "vitoria": mota.desenhar(tela) for obs in obstaculos: obs.desenhar(tela) # HUD texto_nivel = fonte_pequena.render(f"Nível {nivel_atual + 1}/8", True, PRETO) tela.blit(texto_nivel, (20, 20)) texto_dist = fonte_pequena.render(f"Distância: {int(distancia)}", True, PRETO) tela.blit(texto_dist, (20, 60)) # Mensagens de estado if estado == "game_over": msg = fonte.render("GAME OVER", True, VERMELHO) tela.blit(msg, (LARGURA//2 - msg.get_width()//2, ALTURA//2 - 80)) msg2 = fonte_pequena.render("Pressione R para tentar novamente", True, PRETO) tela.blit(msg2, (LARGURA//2 - msg2.get_width()//2, ALTURA//2 + 20)) elif estado == "nivel_completo": msg = fonte.render(f"NÍVEL {nivel_atual} COMPLETO!", True, (0, 180, 0)) tela.blit(msg, (LARGURA//2 - msg.get_width()//2, ALTURA//2 - 60)) # Volta automaticamente para jogar após 1.2 segundos if time.time() - tempo_mensagem > 1.2: estado = "jogando" elif estado == "vitoria": msg = fonte.render("PARABÉNS! VENCES-TE TODOS OS 8 NÍVEIS!", True, (0, 200, 0)) tela.blit(msg, (LARGURA//2 - msg.get_width()//2, ALTURA//2 - 100)) msg2 = fonte_pequena.render("Pressione ESC para sair", True, PRETO) tela.blit(msg2, (LARGURA//2 - msg2.get_width()//2, ALTURA//2 + 40)) pygame.display.flip() relogio.tick(60)
