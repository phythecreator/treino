import os class AvaliacaoFinal: def __init__(self): self.notas_testes = [] self.peso_testes = 0.60 self.peso_comportamento = 0.20 self.peso_questoes = 0.20 # Parâmetros de comportamento self.pontualidade = 0 self.atitudes_valores = 0 self.empenho = 0 self.nota_questoes_aula = 0 def limpar_tela(self): """Limpa a tela do terminal""" os.system('cls' if os.name == 'nt' else 'clear') def coletar_notas_testes(self): """Coleta as notas dos testes (2, 3 ou 4 testes)""" print("\n" + "="*50) print(" COLETA DE NOTAS DOS TESTES") print("="*50) while True: try: num_testes = int(input("\nQuantos testes você realizou? (2, 3 ou 4): ")) if num_testes in [2, 3, 4]: break else: print("Por favor, escolha 2, 3 ou 4 testes.") except ValueError: print("Entrada inválida. Digite um número.") print("\n" + "-"*40) for i in range(num_testes): while True: try: nota = float(input(f"Nota do {i+1}º teste (0 a 5): ")) if 0 <= nota <= 5: self.notas_testes.append(nota) break else: print("Nota deve estar entre 0 e 5.") except ValueError: print("Digite um valor numérico válido.") self.mostrar_notas_testes() def mostrar_notas_testes(self): """Mostra as notas dos testes coletadas""" print("\n" + "-"*40) print("Notas coletadas:") for i, nota in enumerate(self.notas_testes, 1): print(f" Teste {i}: {nota:.1f}") print("-"*40) def coletar_comportamento(self): """Coleta os 3 parâmetros de comportamento""" print("\n" + "="*50) print(" AVALIAÇÃO DE COMPORTAMENTO") print("="*50) print("(0 = Insatisfatório, 5 = Excelente)\n") # Coletar pontualidade while True: try: self.pontualidade = float(input("Pontualidade (0 a 5): ")) if 0 <= self.pontualidade <= 5: break else: print("Nota deve estar entre 0 e 5.") except ValueError: print("Digite um valor numérico válido.") # Coletar atitudes e valores while True: try: self.atitudes_valores = float(input("Atitudes e Valores (0 a 5): ")) if 0 <= self.atitudes_valores <= 5: break else: print("Nota deve estar entre 0 e 5.") except ValueError: print("Digite um valor numérico válido.") # Coletar empenho while True: try: self.empenho = float(input("Empenho (0 a 5): ")) if 0 <= self.empenho <= 5: break else: print("Nota deve estar entre 0 e 5.") except ValueError: print("Digite um valor numérico válido.") def coletar_questoes_aula(self): """Coleta a nota das questões de aula""" print("\n" + "="*50) print(" QUESTÕES DE AULA") print("="*50) while True: try: self.nota_questoes_aula = float(input("Nota das questões de aula (0 a 5): ")) if 0 <= self.nota_questoes_aula <= 5: break else: print("Nota deve estar entre 0 e 5.") except ValueError: print("Digite um valor numérico válido.") def calcular_media_testes(self): """Calcula a média dos testes""" if not self.notas_testes: return 0 return sum(self.notas_testes) / len(self.notas_testes) def calcular_nota_comportamento(self): """Calcula a média dos 3 parâmetros de comportamento""" return (self.pontualidade + self.atitudes_valores + self.empenho) / 3 def calcular_media_final(self): """Calcula a média final ponderada""" media_testes = self.calcular_media_testes() nota_comportamento = self.calcular_nota_comportamento() media_final = (media_testes * self.peso_testes + nota_comportamento * self.peso_comportamento + self.nota_questoes_aula * self.peso_questoes) return media_final def obter_classificacao(self, media): """Retorna a classificação baseada na média""" if media >= 4.5: return "Excelente! 🎉" elif media >= 3.5: return "Bom! 👍" elif media >= 2.5: return "Regular. 😊" elif media >= 1.5: return "Insuficiente. 😟" else: return "Muito Insuficiente. 😢" def mostrar_resultado(self): """Exibe o resultado final da avaliação""" self.limpar_tela() print("\n" + "="*50) print(" RESULTADO FINAL") print("="*50) # Mostrar notas dos testes print("\n📝 TESTES (60%):") for i, nota in enumerate(self.notas_testes, 1): print(f" Teste {i}: {nota:.1f}") media_testes = self.calcular_media_testes() print(f" Média dos testes: {media_testes:.1f}") print(f" Contribuição (60%): {media_testes * self.peso_testes:.2f}") # Mostrar comportamento print("\n👥 COMPORTAMENTO (20%):") print(f" Pontualidade: {self.pontualidade:.1f}") print(f" Atitudes e Valores: {self.atitudes_valores:.1f}") print(f" Empenho: {self.empenho:.1f}") nota_comportamento = self.calcular_nota_comportamento() print(f" Média do comportamento: {nota_comportamento:.1f}") print(f" Contribuição (20%): {nota_comportamento * self.peso_comportamento:.2f}") # Mostrar questões de aula print("\n📚 QUESTÕES DE AULA (20%):") print(f" Nota: {self.nota_questoes_aula:.1f}") print(f" Contribuição (20%): {self.nota_questoes_aula * self.peso_questoes:.2f}") # Calcular e mostrar média final media_final = self.calcular_media_final() classificacao = self.obter_classificacao(media_final) print("\n" + "="*50) print(f"🎯 MÉDIA FINAL: {media_final:.1f} / 5.0") print(f"📊 CLASSIFICAÇÃO: {classificacao}") print("="*50) # Dicas baseadas na média print("\n💡 DICAS:") if media_final >= 4.0: print(" Parabéns! Continue mantendo o excelente desempenho!") elif media_final >= 3.0: print(" Bom trabalho! Você está no caminho certo.") elif media_final >= 2.0: print(" Você precisa se dedicar mais. Revise os conteúdos!") else: print(" É necessário melhorar significativamente. Busque ajuda!") def executar(self): """Executa o programa principal""" self.limpar_tela() print("="*50) print(" SISTEMA DE AVALIAÇÃO FINAL") print("="*50) print("\nEste sistema calcula sua média final considerando:") print("• Testes (60%) - 2, 3 ou 4 testes") print("• Comportamento (20%) - Pontualidade, Atitudes/Valores, Empenho") print("• Questões de Aula (20%)") input("\nPressione ENTER para continuar...") self.coletar_notas_testes() self.coletar_comportamento() self.coletar_questoes_aula() self.mostrar_resultado() print("\n" + "="*50) input("Pressione ENTER para sair...") # Executar o programa if __name__ == "__main__": avaliacao = AvaliacaoFinal() avaliacao.executar()
